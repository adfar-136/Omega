import React, { Component } from 'react'

export default class App extends Component {
  constructor(){
    super()
    this.name =React.createRef()
    this.password = React.createRef()
  }
  submitHandler=(event)=>{
    event.preventDefault()
    console.log(this.name.current.value,this.name.current.placeholder,this.password.current.value)
  }
  render() {
    return (
      <div>
        <h1>Uncontrolled Components</h1>
        <form onSubmit={this.submitHandler}>
          <input type="text" placeholder='enter your name' ref={this.name}/> <br /><br />
          <input type="password" placeholder='enter your password' ref={this.password} /><br /><br />
          <button type='submit'>Submit</button>
        </form>
      </div>
    )
  }
}
